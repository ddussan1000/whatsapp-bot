// src/bot/classifier.ts export function classify(msg: WaMessage): MsgType { if (msg.type === 'image') return 'receipt'; if (msg.type === 'interactive') return 'button_reply'; const text = msg.text?.body?.toLowerCase() ?? ''; const keywords = { price: ['precio','costo','cuánto','vale','valor'], pay: ['pagar','pago','transferir','nequi','bancolombia'], products: ['producto','catálogo','tienes','disponible'], greeting: ['hola','buenas','buen dia','hey'], help: ['ayuda','help','soporte'], }; for (const [type, words] of Object.entries(keywords)) { if (words.some(w => text.includes(w))) return type as MsgType; } return 'free_text'; // → Claude Haiku }

03
OCR + extracción de monto y fecha
// src/receipts/ocr.ts import Tesseract from 'tesseract.js'; import sharp from 'sharp'; export async function extractPaymentData(imgBuffer: Buffer) { // Preprocesar: escala de grises + contraste const processed = await sharp(imgBuffer) .grayscale().normalize().sharpen().toBuffer(); const { data: { text } } = await Tesseract.recognize( processed, 'spa' /* español */ ); // Extraer monto (patrones colombianos) const amountPatterns = [ /(?:valor|monto|total|por)[:\s]*\$?\s*([\d.,]+)/i, /\$\s*([\d]{1,3}(?:[.,]\d{3})+)/, /([\d]{1,3}(?:\.\d{3})+)/, ]; let amount = null; for (const p of amountPatterns) { const m = text.match(p); if (m) { amount = parseFloat(m[1].replace(/\./g,'')); break; } } // Extraer fecha const dateRx = /(\d{1,2})[\/\-\s](\d{1,2})[\/\-\s](\d{2,4})/; const dm = text.match(dateRx); const receiptDate = dm ? new Date(`${dm[3]}-${dm[2]}-${dm[1]}`) : null; // Validar 48 horas const isRecent = receiptDate ? (Date.now() - receiptDate.getTime()) < 48 * 3600000 : false; return { amount, receiptDate, isRecent }; }

Handler completo de comprobante + guardar monto
// src/receipts/handler.ts export async function handleReceipt(msg, phone, state) { const buffer = await downloadFromMeta(msg.image.id); const { amount, receiptDate, isRecent } = await extractPaymentData(buffer); const url = await saveReceipt(buffer, phone); if (!isRecent) { return sendMessage(phone, { type: 'text', text: { body: '⚠️ El comprobante tiene más de 48h. Envía uno reciente.' }}); } if (!amount) { return sendMessage(phone, { type: 'text', text: { body: '⚠️ No pude leer el monto. ¿Puedes enviar foto más clara?' }}); } // ✅ Guardar pago con monto en Supabase await supabase.from('payments').insert({ phone, product: state.product, amount, // ← monto extraído del comprobante receipt_url: url, receipt_date: receiptDate, conversation_id: state.id, }); await setState(phone, { ...state, stage: 'pagó' }); return sendMessage(phone, { type: 'text', text: { body: `✅ Pago de $${amount.toLocaleString('es-CO')} confirmado. ¡Tu pedido está en proceso! 🎉` }}); }
